// index.html already contains this registration logic. Keep this only as a reference snippet.
if ("serviceWorker" in navigator) {
  window.addEventListener("load", async () => {
    try {
      const registration = await navigator.serviceWorker.register("./sw.js?v=210", {
        updateViaCache: "none"
      });
      registration.update?.();

      let reloaded = false;
      navigator.serviceWorker.addEventListener("controllerchange", () => {
        if (reloaded) return;
        reloaded = true;
        location.reload();
      });
    } catch (error) {
      console.warn("Service worker registration failed", error);
    }
  });
}
